192// ======== MAIN BUTTON CALLBACK ========
208static void btnm_event_cb(lv_event_t *e)
209{
210 lv_obj_t *btnm = (lv_obj_t *)lv_event_get_target(e);
211 const char *txt = lv_btnmatrix_get_btn_text(
212 btnm, lv_btnmatrix_get_selected_btn(btnm));
213
214 if (txt == NULL)
215 return;
216
217 if (strcmp(txt, "Forward") == 0 || strcmp(txt, "Backward") == 0)
218 {
219 apply_mode("Straight", txt);
220 }
221 else if (strcmp(txt, "Left") == 0 || strcmp(txt, "Right") == 0)
222 {
223 apply_mode("Sideways", txt);
224 }
225 else if (strcmp(txt, "45") == 0 || strcmp(txt, "135") == 0 || strcmp(txt, "225") == 0 || strcmp(txt, "315") == 0)
226 {
227 apply_mode("Diagonal", txt);
228 }
229 else if (strcmp(txt, "Right Fwd") == 0 || strcmp(txt, "Right Bwd") == 0 || strcmp(txt, "Left Fwd") == 0 || strcmp(txt, "Left Bwd") == 0)
230 {
231 apply_mode("Pivot", txt);
232 }
233 else if (strcmp(txt, "Front Right") == 0 || strcmp(txt, "Front Left") == 0 || strcmp(txt, "Rear Right") == 0 || strcmp(txt, "Rear Left") == 0)
234 {
235 apply_mode("Pivot Sideways", txt);
236 }
237 else if (strcmp(txt, "CCW") == 0 || strcmp(txt, "CW") == 0)
238 {
239 apply_mode("Rotate", txt);
240 }
241 else if (strcmp(txt, "Prgm Car") == 0)
242 {
243 btn_Run_Program_event_cb(e);
244 }
245}

308static void toggle_fwdrev_cb(lv_event_t *e)
309{
310 lv_obj_t *btn = (lv_obj_t *)lv_event_get_target(e);
311 lv_obj_t *label = lv_obj_get_child(btn, 0);
312
313 const char *txt = lv_label_get_text(label);
314
315 if (strcmp(txt, "Fwd") == 0) // If txt is "Fwd", change to "Rev" and assign motor_direction 1
316 {
317 lv_label_set_text(label, "Rev");
318 motor_direction = 1;
319 }
320 else // If txt is "Rev", change to "Fwd" and assign motor_direction -1
321 {
322 lv_label_set_text(label, "Fwd");
323 motor_direction = -1;
324 }
325
326 Serial.print("Mode: ");
327 Serial.println(lv_label_get_text(label));
328 Serial.print("Direction set to ");
329 Serial.println(motor_direction);

414void create_ui()
415{
416 screen_buttons = lv_obj_create(NULL);
417 create_button_matrix(screen_buttons);
418 create_test_button_matrix(screen_buttons);
419
420 // ===== Speed Sliders =====
421 const char *labels[] = {"FR", "FL", "RR", "RL"};
422 int y_positions[] = {20, 60, 100, 140};
423
424 for (int i = 0; i < 4; i++)
425 {
426 // Create label
427 lv_obj_t *label = lv_label_create(screen_buttons);
428 lv_label_set_text(label, labels[i]);
429 lv_obj_align(label, LV_ALIGN_TOP_RIGHT, -230, y_positions[i]);
430
431 // Create slider
432 lv_obj_t *slider = lv_slider_create(screen_buttons);
433 lv_slider_set_range(slider, 0, 255);
434 lv_obj_set_size(slider, 150, 20);
435 lv_obj_align_to(slider, label, LV_ALIGN_OUT_RIGHT_MID, 15, 0);
436
437 // Create value label
438 lv_obj_t *value_label = lv_label_create(screen_buttons);
439 char buf[4];
440 snprintf(buf, sizeof(buf), "%d", lv_slider_get_value(slider));
441 lv_label_set_text(value_label, buf);
442 lv_obj_align_to(value_label, slider, LV_ALIGN_OUT_RIGHT_MID, 15, 0);
443
444 // Add event callback that updates the value label
445 lv_obj_add_event_cb(
446 slider, [](lv_event_t *e)
447 {
448 lv_obj_t* s = (lv_obj_t*)lv_event_get_target(e);
449 lv_obj_t* val_lbl = (lv_obj_t*)lv_event_get_user_data(e);
450 char buf[4];
451 snprintf(buf, sizeof(buf), "%d", lv_slider_get_value(s));
452 lv_label_set_text(val_lbl, buf); },
453 LV_EVENT_VALUE_CHANGED, value_label);
454
455 // Add slider event callback to update speed
456 lv_obj_add_event_cb(slider, slider_event_cb, LV_EVENT_VALUE_CHANGED, (void *)labels[i]);
457 }
458
459 // ===== Fwd/Rev Toggle Button =====
460 lv_obj_t *toggle_FwdRev = lv_btn_create(screen_buttons);
461 lv_obj_set_size(toggle_FwdRev, 80, 40);
462 lv_obj_align(toggle_FwdRev, LV_ALIGN_TOP_RIGHT, -20, 180);
463
464 lv_obj_t *toggle_label = lv_label_create(toggle_FwdRev);
465 lv_label_set_text(toggle_label, "Fwd");
466 lv_obj_center(toggle_label);
467
468 // Set direction
469 lv_obj_add_event_cb(toggle_FwdRev, toggle_fwdrev_cb, LV_EVENT_CLICKED, NULL);
470}
